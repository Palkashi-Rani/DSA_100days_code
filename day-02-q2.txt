Q2~ Write a C program to delete the element at a given 1-based position pos from an array of n integers. Shift remaining elements to the left.


#include <stdio.h>

int main() {
    int a[50], size, pos, i;

   
    scanf("%d", &size);

    
    for (i = 0; i < size; i++) {
        scanf("%d", &a[i]);
    }

    scanf("%d", &pos);

    if (pos <= 0 || pos > size) {
        printf("Invalid position\n");
        return 0;
    }

    for (i = pos - 1; i < size - 1; i++) {
        a[i] = a[i + 1];
    }
    size--;

    for (i = 0; i < size; i++) {
        printf("%d", a[i]);
        if (i != size - 1) {
            printf(" ");
        }
    }

    return 0;
}
